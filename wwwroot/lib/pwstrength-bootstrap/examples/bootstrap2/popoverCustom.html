jQuery(document).ready(function () {
    "use strict";
    var options = {};
    options.ui = {
        showPopover: true,
        showErrors: true,
        bootstrap2: true,
        popoverError: function (errors) {
            "use strict";
            var message = "<div><ul class='error-list' style='margin-bottom: 0;'>";

            /*jslint unparam: true*/
            jQuery.each(errors, function (idx, err) {
                message += "<li>" + err + "</li>";
            });
            message += "</ul></div>";
            return message;
        }
    };
    $(':password').pwstrength(options);
});
